%------------ Jacobi Content ---------------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.3\textwidth}
    \textbf{Convergence} \\
    $\mathbf{A} = \mathbf{L}+\mathbf{D}+\mathbf{U}, \mathbf{A}x = \mathbf{b}$ \\
    $(\mathbf{L}+\mathbf{D}+\mathbf{U}) x = \mathbf{b}$ \\
    $\mathbf{D}x = -(\mathbf{L}+\mathbf{D}^{-1}\mathbf{U})x + \mathbf{b}$ \\
    $x = -\mathbf{D}^{-1}(\mathbf{L}+\mathbf{U})x + \mathbf{D}^{-1}\mathbf{b}$ \\
    $x = \mathbf{T}x + \mathbf{b}', x^0 = 0, x^1 = \mathbf{b}'$ \\
    $x^2 = (\mathbf{T} + \mathbf{I}) \mathbf{b}', x^3 = (\mathbf{T}^2 + \mathbf{T} + \mathbf{I}) \mathbf{b}'$ \\
    Induction $x^k = (\mathbf{T}^{k-1} + ... + \mathbf{T} + \mathbf{I}) \mathbf{b}'$ \\
    Converges if $r \left ( \mathbf{D}^{-1}(\mathbf{L}+\mathbf{U}) \right ) < 1$ (Neumann series) \\
     - diagonally dominant converges\\
    \hline \hfill
\begin{verbatim}
    n = size(A);
    residuals = zeros(1, n_iter);
    x = x_0;
    for i = 1 : n_iter
        x_next = x;
        for j = 1 : n
            x_next(j, 1) = (
                b(j, 1) 
                - A(j, [1:j-1, j+1:n]) 
                    * x([1:j-1, j+1:n], 1)
                ) / A(j, j);
        end
        x = x_next;
        residuals(1, i) = norm(A * x - b);
    end
    solution = x;
\end{verbatim}
    \end{minipage}
};
%------------ Jacobi Header ---------------------
\node[fancytitle, right=10pt] at (box.north west) {Jacobi};
\end{tikzpicture}

%------------ Gauss-Seidel Content ---------------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.3\textwidth}
    \textbf{Convergence} \\
    $x = -\mathbf{D}^{-1}(\mathbf{L}+\mathbf{U})x + \mathbf{D}^{-1}\mathbf{b}$ \\
    $x = -\mathbf{D}^{-1}\mathbf{L}x -\mathbf{D}^{-1}\mathbf{U}x + \mathbf{D}^{-1}\mathbf{b}$ \\
    $x^{k+1} = -\mathbf{D}^{-1}\mathbf{L}x^{k+1} -\mathbf{D}^{-1}\mathbf{U}x^k + \mathbf{b}'$ \\
    $(\mathbf{I} + \mathbf{D}^{-1}\mathbf{L})x^{k+1} = -\mathbf{D}^{-1}\mathbf{U}x^k + \mathbf{b}'$ \\
    $(\mathbf{I} + \mathbf{D}^{-1}\mathbf{L})^{-1} \mathbf{D}^{-1} = \left [ \mathbf{D}(\mathbf{I} + \mathbf{D}^{-1}\mathbf{L})\right ] ^{-1}$ \\
    $ = (\mathbf{D} + \mathbf{L})^{-1}, x = \mathbf{T}x + \mathbf{b}''$ \\
    $\mathbf{T} = -(\mathbf{D}+\mathbf{L})^{-1} \mathbf{U}, \mathbf{b}'' = (\mathbf{D}+\mathbf{L})^{-1} \mathbf{b}'$ \\
    Converges if $r \left ( (\mathbf{D}+\mathbf{L})^{-1} \mathbf{U}) \right ) < 1$ (Neumann series) \\
     - diagonally dominant converges\\
    \hline \hfill
\begin{verbatim}
    n = size(A);
    residuals = zeros(1, n_iter);
    x = x_0;
    for i = 1 : n_iter
        x_next = x;
        for j = 1 : n
            x_next(j, 1) = (
                b(j, 1) 
                - A(j, 1 : j - 1) 
                    * x_next(1 : j-1, 1) 
                - A(j, j+1 : n) 
                    * x(j+1 : n, 1)
                ) / A(j, j);
        x = x_next;
        residuals(1, i) = norm(A * x - b);
    solution = x;
\end{verbatim}
    \end{minipage}
};

\node[fancytitle, right=10pt] at (box.north west) {Gauss-Seidel};
\end{tikzpicture}

%------------ Givens Content ---------------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.3\textwidth}

\begin{verbatim}
[m, n] = size(A);
Q = eye(m);
R = A;
for i = 1 : n
    for j = i + 1 : m
        x1 = R(i, i);
        x2 = R(j, i);
        if x2 != 0
            c = x1 / sqrt(x1^2 + x2^2);
            s = - x2 / sqrt(x1^2 + x2^2);
            G = eye(m);
            G(i, i) = c;
            G(j, j) = c;
            G(i, j) = -s;
            G(j, i) = s;
            R = G * R
            Q = Q * transpose(G)
\end{verbatim}

%------------ Householder Content ---------------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.3\textwidth}
    $ \mathbf{H} \mathbf{a} = \mathbf{b}, \mathbf{H} \mathbf{b} = \mathbf{a}, \mathbf{u}||\mathbf{a}-\mathbf{b}|| = \mathbf{a} - \mathbf{b} $\\
    $ \mathbf{H} = \mathbf{I} - 2 \mathbf{u}\mathbf{u}^{*}$ \\
    $ \mathbf{H}\mathbf{H}^\mathsf{T} = \mathbf{H}^\mathsf{T}\mathbf{H} = \mathbf{I}$ orthogonal \\
    \hline \hfill
\begin{verbatim}
[m, n] = size(A);
Q = eye(m);
R = A;

for i = 1 : n
    u = R(:, i);
    gamma = sqrt(sum(R(i:m, i).^2));
    if i > 1
        u(1:i-1, :) = 0;
    u(i) = R(i, i) + sign(R(i, i)) * gamma;
    u = u / sqrt(2 * gamma * 
                 (gamma + abs(R(i, i))));
    H = I - 2 * (u * u');
    Q = Q * H;
    R = H * R;
\end{verbatim}
    \end{minipage}
};
